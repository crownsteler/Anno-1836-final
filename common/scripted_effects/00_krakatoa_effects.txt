# Reusable Tsunami
# scope = strategic region
tsunami_effect = {
	custom_description = {
		text = "tsunami_effect"
		subject = this
		value = $value$


		every_scope_state = {
			limit = {
				is_coastal = yes
			}
			state_region = {
				add_to_temporary_list = tsunami_list
			}
		}
		every_in_list = {
			list = tsunami_list

			set_variable = { # We want Tsunamis to ignore and reset the natural disaster cooldown
				name = natural_disaster_cooldown
				days = normal_modifier_time
			}
			set_variable = {
				name = is_tsunami_state
				days = normal_modifier_time
			}
			remove_from_list = tsunami_list
			if = {
				limit = {
					NOT = {
						has_harvest_condition = tsunami
					}
				}
				start_harvest_condition = tsunami
			}
			add_devastation = $value$
		}
	}
}
