movement_centralist = {
	category = movement_category_ideological

	ideology = ideology_centralist_movement	
	character_ideologies = {
		ideology_centralist_leader
		ideology_traditionalist
		ideology_royalist
		ideology_authoritarian
		ideology_modernizer_leader
	}
	
	disband_trigger = {
		NOT = { has_variable = cr_centralist_federalist_struggle }

		always = no

	creation_weight = {
		value = 100
	}	
	on_created = {
		#set_bolstering = yes
	}
	
	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_centralist_leader
			has_ideology = ideology:ideology_traditionalist
			has_ideology = ideology:ideology_royalist
			has_ideology = ideology:ideology_authoritarian
			has_ideology = ideology:ideology_modernizer_leader
		}
	}
	character_support_weight = {
		value = 100
		if = {
			limit = {
				scope:interest_group ?= {
					OR = {
						is_interest_group_type = ig_devout
						is_interest_group_type = ig_landowners
						is_interest_group_type = ig_industrialists
					}
				}
			}
			multiply = {
				value = 2
			}
		}
		if = {
			limit = {
				scope:interest_group ?= {
					is_interest_group_type = ig_petty_bourgeoisie
				}
			}
			multiply = {
				value = 1.5
			}
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_centralist_leader
			}
			multiply = {
				value = 100
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_federalist_leader
			}
			multiply = {
				value = 0
			}
		}
		else_if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_royalist
					has_ideology = ideology:ideology_authoritarian
				}
			}
			multiply = {
				value = 5
			}
		}
	}
	
	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_ruler_popularity
		movement_support_capitalists
		movement_support_aristocrats
		movement_support_clergymen
		movement_support_officers
	}	

	pop_support_weight = {	
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}
		add = {
			desc = "URBAN_STATE"
			value = 4

		#if = {	#Handled via the modifiers themselves

		#	}
		#}
		
		if = {
			limit = {
				is_pop_type = clergymen
			}
			add = {
				value = 25
				desc = "POP_CLERGYMEN"
			}
		}
		else_if = {
			limit = {
				is_pop_type = aristocrats
			}
			add = {
				value = 30
				desc = "POP_ARISTOCRATS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = capitalists
			}
			add = {
				value = 20
				desc = "POP_CAPITALISTS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = officers
			}
			add = {
				value = 30
				desc = "POP_OFFICERS"
			}
		}
		#else_if = {
		#	limit = {
		#		is_pop_type = machinists
		#	}
		#	add = {
		#		value = 25
		#		desc = "POP_MACHINISTS"
		#	}

		#	}

		#		value = 15
		#		desc = "POP_FARMERS"
		#	}
		#}

		#	limit = {
		#		is_pop_type = laborers
		#	}
		#	add = {
		#		value = 15
		#		desc = "POP_LABORERS"
		#	}
		#}
		else_if = {
			limit = {
				is_pop_type = academics
			}
			subtract = 10
		}
		else_if = {
			limit = {
				is_pop_type = clerks
			}
			subtract = 5
		}
		else_if = {
			limit = {
				is_pop_type = engineers
			}
			subtract = 5
		}

		#Centralists care about strongman rulers, federalists about government legitimacy.
		if = {
			limit = {
				owner ?= {
					has_law_or_variant = law_type:law_hereditary_bureaucrats
				}
			}
			multiply = {
				add = owner.ruler.popularity
				multiply = 0.01
				add = 1
				min = 0.2
				max = 1.8
				desc = "RULER_POPULARITY"
			}
		}
		else_if = {
			multiply = {
				add = owner.ruler.popularity
				multiply = -0.01
				add = 1
				min = 0.2
				max = 1.8
				desc = "RULER_IMPOPULARITY"
			}
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.30
		}
		
		weight = {
			value = 100
		}
		
		state_weight = {
			value = 1
			if = {
			   limit = { 
					has_modifier = cr_mexico_centralist_sympathies
			   }
				multiply = 2.0
			}	
		}
	}

	secession = {
