je_pacify_the_steppes_sent_expedition_button = {
	name = "je_pacify_the_steppes_sent_expedition_button"
	desc = "je_pacify_the_steppes_sent_expedition_button_desc"
	
	visible = {   
		has_variable = cr_steppe_first_expedition_has_returned
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				authority >= 100
			}
			add = 10
		}
		if = {
			limit = {
				is_at_war = yes
			}
			add = -10
		}
		if = {
			limit = {
				is_diplomatic_play_committed_participant = yes
			}
			add = -10
		}
	}
	
	possible = {
		institution_investment_level = {
			institution = institution_colonial_affairs
			value >= 1	#Could perhaps be raised to level 2 for the player? Not AI as AI wouldn't be able to strategise around this.
		}
		custom_tooltip = {
			text = je_pacify_the_steppes_know_where_kenesary_is
			NOT = { has_variable = cr_steppe_kenesary_in_hiding } #Removed via vanilla events kazakhstan_events.4-8
		}
		custom_tooltip = {
			text = je_pacify_the_steppes_no_active_expedition
			NOT = { has_modifier = cr_steppe_punitive_expedition }
		}
		custom_tooltip = {
			text = je_pacify_the_steppes_no_recent_expedition
			NOT = { 
				has_modifier = cr_steppe_punitive_expedition
				has_variable = cr_steppe_had_recent_expedition #Added after each expedition result with duration of 1 year.
			} #Added after each expedition result with duration of 1 year.
		}
		is_at_war = no
		is_diplomatic_play_committed_participant = no
	}

	effect = {
		cr_steppe_punitive_expedition_effect = yes
	}
}
