je_seminole_war_containment_policy = {
	name = "je_seminole_war_containment_policy"
	desc = "je_seminole_war_containment_policy_desc"

	visible = { 
		always = yes
	}

	ai_chance = {
		base = 0
		if = {
			add = 1
			limit = { 
				has_variable = seminole_war_ai_select_containment_policy
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			text = je_seminole_war_policy_cooldown
			NOT = {
				has_variable = seminole_war_policy_cooldown
			}
		}
		ruler = {
			NOT = { has_trait = reckless }
		}
		custom_tooltip = {
			text = je_seminole_war_policy_already_active
			NOT = {
				has_variable = seminole_war_containment_policy
			}
		}
	}

	effect = {
		custom_tooltip = {
			text = je_seminole_war_containment_policy_tt
			remove_variable = seminole_war_forward_policy
			remove_variable = seminole_war_diplomatic_policy
			set_variable = seminole_war_containment_policy
			
			set_variable = {
				name = seminole_war_policy_cooldown
				days = 365
			}
		}
		scope:flarda = {
			if = {
				limit = { has_modifier = seminole_war_diplomatic_policy }
				remove_modifier = seminole_war_diplomatic_policy
			}
			if = {
				limit = { has_modifier = seminole_war_forward_policy }
				remove_modifier = seminole_war_forward_policy
			}
			add_modifier = seminole_war_containment_policy
		}
		remove_modifier = seminole_war_expense_mod
		add_modifier = {
			name = seminole_war_expense_mod
			multiplier = 0.5
		}

	}
}
